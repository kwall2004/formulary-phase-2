573  public static ExternalLoginData FromIdentity(ClaimsIdentity identity)
574  {
575  if (identity == null)

579 
580  Claim providerKeyClaim = identity.FindFirst(ClaimTypes.NameIdentifier);
581 
582  if (providerKeyClaim == null || String.IsNullOrEmpty(providerKeyClaim.Issuer)
583  || String.IsNullOrEmpty(providerKeyClaim.Value))
584  {
585  return null;
586  }

618  public static string Generate(int strengthInBits)
619  {
620  const int bitsPerByte = 8;
621 
622  if (strengthInBits % bitsPerByte != 0)
623  {
624  throw new ArgumentException("strengthInBits must be evenly divisible by 8.", "strengthInBits");
625  }
626 
627  int strengthInBytes = strengthInBits / bitsPerByte;
628 
629  byte[] data = new byte[strengthInBytes];
